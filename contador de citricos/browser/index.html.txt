    <script>
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');
        let video = null;

        // Inicializar la cámara del celular (cámara trasera)
        async function startCamera() {
            video = document.createElement('video');
            const stream = await navigator.mediaDevices.getUserMedia({
                video: { facingMode: "environment" }, // Fuerza la cámara trasera
                audio: false
            });
            video.srcObject = stream;
            video.play();
            
            video.onloadedmetadata = () => {
                canvas.width = video.videoWidth;
                canvas.height = video.videoHeight;
                processFrame();
            };
        }

        // Procesar cada cuadro de video
        async function processFrame() {
            // Dibujar el video en el lienzo (canvas)
            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

            // Obtener los datos de los píxeles para el modelo
            const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
            
            // --- INFERENCIA ---
            // Llamamos a la biblioteca de Edge Impulse pasándole la imagen
            let classifier = new EdgeImpulseClassifier();
            await classifier.init();
            let result = await classifier.classify(imgData.data);

            // Reiniciamos los contadores para este fotograma (conteo estático)
            let totalLimones = 0;
            let totalNaranjas = 0;

            if (result.bounding_boxes) {
                result.bounding_boxes.forEach(box => {
                    // Filtramos detecciones con baja seguridad (menor al 50%)
                    if (box.value > 0.5) {
                        
                        // Dibujar recuadro de detección
                        ctx.strokeStyle = box.label === 'limon' ? '#f1c40f' : '#e67e22';
                        ctx.lineWidth = 4;
                        ctx.strokeRect(box.x, box.y, box.width, box.height);

                        // Contar según la etiqueta del modelo
                        if (box.label === 'limon') {
                            totalLimones++;
                        } else if (box.label === 'naranja') {
                            totalNaranjas++;
                        }
                    }
                });
            }

            // Actualizar los números en la pantalla
            document.getElementById('txt-limones').innerText = totalLimones;
            document.getElementById('txt-naranjas').innerText = totalNaranjas;

            // Volver a procesar el siguiente fotograma
            requestAnimationFrame(processFrame);
        }

        // Iniciar todo al cargar la página
        window.onload = startCamera;
    </script>
